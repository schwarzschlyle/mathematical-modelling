1-D harmonic (linear) oscillator with viscous friction has equation

\begin{equation}
    \frac{\text{d}^2x}{\text{d}t^2} + \kappa\frac{\text{d}x}{\text{d}t} +\omega_0^2x=0
\end{equation}

\lkg{1. Reconcile the following solution with analytical methods}

\begin{equation}
    x(t) = e^{at}\left(x_0\text{ cos}(\omega t)+\left(\frac{p_0}{m\omega}\right)\text{sin}(\omega t)\right)
\end{equation}


\lkg{2. Determine $\omega(x(0),x'(0))$, $x_0(x(0),x'(0))$, and $p_0(x(0),x'(0))$} 


\lkg{3. Plot phase portraits for $x(t)$ and $p(t)$ for several values of $p_0$ and $\omega_0=0.8$}
